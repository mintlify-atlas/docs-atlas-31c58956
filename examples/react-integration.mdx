## Overview

This guide shows you how to integrate Nanolytics into React applications, including initialization, custom hooks, component lifecycle tracking, and best practices.

## Basic Setup

### Initialize in App Component

Initialize Nanolytics once when your app starts:

```jsx App.jsx
import { useEffect } from 'react';
import { init, record } from 'nanolytics';

function App() {
  useEffect(() => {
    // Define submission function
    const submitAnalytics = async (userInfo, events) => {
      const response = await fetch('/api/analytics', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
        },
        body: JSON.stringify({ 
          state: userInfo, 
          data: events 
        }),
      });
      
      if (!response.ok) {
        throw new Error('Failed to submit analytics');
      }
    };

    // Initialize only once
    init(submitAnalytics, { maxEventsLimitUntilSubmit: 150 });
  }, []); // Empty dependency array ensures this runs once

  return (
    <div className="App">
      {/* Your app components */}
    </div>
  );
}

export default App;
```

## Custom Hooks

### useAnalytics Hook

Create a custom hook for easy event tracking:

```jsx hooks/useAnalytics.js
import { useCallback } from 'react';
import { record, commit } from 'nanolytics';

export const useAnalytics = () => {
  const track = useCallback((event, context, data) => {
    record(event, context, data);
  }, []);

  const trackClick = useCallback((buttonName, additionalData = {}) => {
    record('ButtonClicked', window.location.pathname, {
      button: buttonName,
      ...additionalData,
    });
  }, []);

  const trackPageView = useCallback((pageName, additionalData = {}) => {
    record('PageView', pageName, {
      title: document.title,
      ...additionalData,
    });
  }, []);

  const endSession = useCallback(() => {
    commit();
  }, []);

  return {
    track,
    trackClick,
    trackPageView,
    endSession,
  };
};
```

## Advanced Patterns

### Track Search Functionality

```jsx components/SearchBar.jsx
import { useState, useCallback } from 'react';
import { record } from 'nanolytics';
import { debounce } from 'lodash';

function SearchBar() {
  const [query, setQuery] = useState('');
  const [results, setResults] = useState([]);

  const trackSearch = useCallback(
    debounce((searchQuery, resultCount) => {
      record('SearchPerformed', 'Header', {
        query: searchQuery,
        resultCount,
        timestamp: Date.now(),
      });
    }, 1000),
    []
  );

  const handleSearch = async (e) => {
    const searchQuery = e.target.value;
    setQuery(searchQuery);

    if (searchQuery.length > 2) {
      const searchResults = await performSearch(searchQuery);
      setResults(searchResults);
      trackSearch(searchQuery, searchResults.length);
    }
  };

  return (
    <input 
      type="search" 
      value={query}
      onChange={handleSearch}
      placeholder="Search..."
    />
  );
}
```

## Error Tracking

### Error Boundary with Analytics

```jsx components/ErrorBoundary.jsx
import React from 'react';
import { record } from 'nanolytics';

class ErrorBoundary extends React.Component {
  constructor(props) {
    super(props);
    this.state = { hasError: false };
  }

  static getDerivedStateFromError(error) {
    return { hasError: true };
  }

  componentDidCatch(error, errorInfo) {
    // Track error
    record('ErrorCaught', 'ErrorBoundary', {
      error: error.toString(),
      errorInfo: errorInfo.componentStack,
      timestamp: Date.now(),
    });
  }

  render() {
    if (this.state.hasError) {
      return <h1>Something went wrong.</h1>;
    }

    return this.props.children;
  }
}

export default ErrorBoundary;
```

### Track Navigation

## TypeScript Support

### Type-Safe Analytics

```typescript hooks/useAnalytics.ts
import { useCallback } from 'react';
import { record, commit } from 'nanolytics';
import type { Event } from 'nanolytics';

interface TrackEventData {
  [key: string]: any;
}

export const useAnalytics = () => {
  const track = useCallback(
    (event: string, context?: string, data?: TrackEventData) => {
      record(event, context, data);
    },
    []
  );

    (buttonName: string, additionalData?: TrackEventData) => {
      record('ButtonClicked', window.location.pathname, {

        ...additionalData,
      });
    },
    []
  );

  return { track, trackClick };
};
```

## Best Practices

<AccordionGroup>
  <Accordion title="Initialize Once">
    Always initialize Nanolytics once in your root component using `useEffect` with an empty dependency array to prevent multiple initializations.
  </Accordion>
  
  <Accordion title="Use Custom Hooks">
    Create custom hooks like `useAnalytics` to encapsulate tracking logic and make it reusable across components.
  </Accordion>
  
  <Accordion title="Memoize Callbacks">
    Use `useCallback` for tracking functions to prevent unnecessary re-renders and maintain referential equality.
  </Accordion>
  
  <Accordion title="Track Meaningful Events">
    Focus on tracking events that provide business value: user actions, conversions, errors, and performance metrics.
  </Accordion>
  
  <Accordion title="Cleanup on Unmount">
    Use the cleanup function in `useEffect` to track component unmount or cancel pending operations.
  </Accordion>
</AccordionGroup>
